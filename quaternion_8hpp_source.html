479 template< typename T>
480 constexpr quaternionU<T> interpolation(const quaternionU<T>& q_start, const quaternionU<T>& q_end, double t, InterpType interp_type = InterpType::LERP) noexcept{
481 switch (interp_type)
482 {
483 case InterpType::LERP:{
484 return q_start * (1.0 - t) + q_end * t;
485 }
486 case InterpType::SLERP:{
487 auto omega = dot(q_start, q_end);
488 return ((std::sin(1 - t) * omega) / std::sin(omega)) * q_start + (sin(t * omega) / sin(omega)) * q_end;
489 }
490 default: {
491 return q_start;
492 }
493 }

496 template< typename T>
497 constexpr auto quatToRotation(const quaternionU<T>& q_in) noexcept{
498
499 auto a0 = pow(q_in.w(), 2);
500 auto a1 = pow(q_in.x(), 2);
501 auto a2 = pow(q_in.y(), 2);
502 auto a3 = pow(q_in.z(), 2);
503
504 auto a4 = q_in.w() * q_in.x();
505 auto a5 = q_in.w() * q_in.y();
506 auto a6 = q_in.w() * q_in.z();
507
508 auto a7 = q_in.x() * q_in.y();
509 auto a8 = q_in.x() * q_in.z();
510
511 auto a9 = q_in.y() * q_in.z();
512
513 std::array<T, 9> R = { 2 * (a0 + a1) - 1, 2 * (a7 - a6), 2 * (a8 + a5),
514 2 * (a7 + a6), 2 * (a0 + a2) - 1, 2 * (a9 - a4),
515 2 * (a8 - a5), 2 * (a9 + a4), 2 * (a0 + a3) - 1};
516
517 return R;
518 }
